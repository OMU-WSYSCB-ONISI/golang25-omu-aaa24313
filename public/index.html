package main import ( "fmt" "net/http" "net/http/httputil" ) func main() { const url = "https://onisi.github.io/" req, _ := http.NewRequest("GET", url, nil) dump, _ := httputil.DumpRequestOut(req, true) fmt.Printf("%s", dump) client := new(http.Client) resp, err := client.Do(req) if err != nil { fmt.Println(err) return } defer resp.Body.Close() dumpResp, _ := httputil.DumpResponse(resp, true) fmt.Printf("%s", dumpResp) } Sample Test File over Golang http.FileServer
